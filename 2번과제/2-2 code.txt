#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
    int answer = 0;
    int m = board[0].size();
    
    for (int i = 0; i < board.size() && i <= k; ++i) {
        int maxj = min(m - 1, k - i);
        for (int j = 0; j <= maxj; ++j) answer += board[i][j];
    }
    
    return answer;
}
