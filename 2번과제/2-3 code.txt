#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    int x_1 = 0, x_2 = 0, y_1 = 0, y_2 = 0;
    
    x_1 = dots[0][0];
    y_1 = dots[0][1];
    for(int i=0; i<4; ++i) {
        if(x_1 != dots[i][0]) x_2 = dots[i][0];
        if(y_1 != dots[i][1]) y_2 = dots[i][1];
    }

    answer = abs(x_1 - x_2) * abs(y_1 - y_2);
    
    return answer;
}
